```javascript
function normalizeDisplayTime(label) {
    if (label === null || label === undefined) return label;
    if (typeof label !== 'string') return label;
    // Strip HTML tags first (like sampler.html does implicitly)
    const withoutHTML = label.replace(/<[^>]*>/g, '');
    const cleaned = withoutHTML
        .replace(/[\u00A0\u2007\u202F]/g, ' ')
        .replace(/\s+/g, ' ')
        .trim();
    if (cleaned === '' || cleaned.toUpperCase() === 'TBD') return 'TBD';
    const timeMatch = cleaned.match(/^(\d{1,2}):(\d{2})\s*(AM|PM)$/i);
    if (timeMatch) {
        let hour = parseInt(timeMatch[1], 10);
        const minutes = timeMatch[2];
        const meridiem = timeMatch[3].toUpperCase();
        if (hour === 12) hour = 0;
        if (meridiem === 'PM') hour += 12;
        return `${hour.toString().padStart(2, '0')}:${minutes}`;
    }
    return cleaned.toUpperCase();
}
```
**Note:** DOES strip HTML tags first.
